Physics webR Lab スニペット集

コピペしてすぐ使えるコード片と、関数の早見表（チートシート）です。コメント（#）は日本語、グラフ内のラベルは英語にしています（webR では日本語が文字化けするため）。

関数チートシート

データを扱う

やりたいこと 書き方
表をデータフレームで受け取る df <- get.input()
シート名 / 番号で受け取る get.input("Sheet2") / get.input(2)
先頭数行を見る / 要約する head(df) / summary(df)
1 列を取り出す df$length_m
列名をまとめて指定 cols <- c("a_s", "b_s", "c_s")
新しい列を足す df$mean_s <- rowMeans(df[cols], na.rm = TRUE)
行ごとの標準偏差 df$sd_s <- apply(df[cols], 1, sd, na.rm = TRUE)
欠損値の行を取り除く df <- subset(df, !is.na(length_m))
条件で行をしぼる subset(df, material != "")

グラフを描く

やりたいこと 書き方
散布図 plot(df$x, df$y, pch = 19)
線グラフ plot(df$x, df$y, type = "l")
点と線の両方 plot(df$x, df$y, type = "o")
棒グラフ barplot(df$mean, names.arg = df$label)
既存の図に点を重ねる points(df$x, df$y2)
既存の図に線を重ねる lines(x, y)
直線を重ねる abline(fit) / abline(h = 0)
誤差棒 arrows(x, y-e, x, y+e, angle = 90, code = 3, length = 0.04)
凡例 legend("topleft", legend = c(...), pch = c(...), col = c(...))
余白・軸を整える（独自） set.paper_style()
補助グリッド（独自） add.paper_guides()

よく使うオプション

オプション 意味 例
pch 点の形 19=●, 17=▲, 15=■, 1=○
col 色 "firebrick", "#2563eb"
cex 点・文字の大きさ 1.2
lwd 線の太さ 2
lty 線の種類 1=実線, 2=破線
xlab/ylab 軸ラベル（英語・単位つき） "Time / s"
main タイトル "Free fall"
xlim/ylim 軸の範囲 ylim = c(0, 3)

スニペット集

平均と標準偏差を求める

df <- get.input()
cols <- c("period_1_s", "period_2_s", "period_3_s")   # 繰り返し測定の列
df$mean_s <- rowMeans(df[cols], na.rm = TRUE)          # 行ごとの平均
df$sd_s   <- apply(df[cols], 1, sd, na.rm = TRUE)      # 行ごとのばらつき
head(df)

誤差棒つき散布図

set.paper_style()
plot(df$x_value, df$y_mean,
  pch = 19, cex = 1.1,
  xlab = "X / unit", ylab = "Y / unit"
)
add.paper_guides()
arrows(
  df$x_value, df$y_mean - df$y_sd,   # 下端
  df$x_value, df$y_mean + df$y_sd,   # 上端
  angle = 90, code = 3, length = 0.04
)

直線あてはめ（線形回帰）

fit <- lm(y_mean ~ x_value, data = df)   # y を x で説明する直線
abline(fit, col = "#374151", lwd = 2)    # 直線を重ねる
summary(fit)                             # 傾き・切片・決定係数を確認

原点を通る直線にしたいときは lm(y ~ 0 + x, data = df) とします。

2 次曲線をあてはめる

df <- subset(get.input(), !is.na(time_s))
fit <- lm(position_m ~ time_s + I(time_s^2), data = df)   # 2次の項を足す

set.paper_style()
plot(df$time_s, df$position_m, pch = 19,
  xlab = "Time / s", ylab = "Position / m")

# 曲線はなめらかな x を作り、predict した値を線でつなぐ
x_grid <- seq(min(df$time_s), max(df$time_s), length.out = 200)
lines(x_grid, predict(fit, newdata = data.frame(time_s = x_grid)),
  col = "#374151", lwd = 2)

複数の系列を重ねて凡例をつける

df <- subset(get.input(), !is.na(time_s))
y_max <- max(c(df$series_a, df$series_b), na.rm = TRUE)

set.paper_style()
plot(df$time_s, df$series_a, pch = 19, col = "firebrick",
  ylim = c(0, y_max),
  xlab = "Time / s", ylab = "Value / unit")
points(df$time_s, df$series_b, pch = 17, col = "#2563eb")   # 2本目を重ねる

legend("topleft",
  legend = c("Series A", "Series B"),
  pch = c(19, 17), col = c("firebrick", "#2563eb"), bty = "n")

時系列を線グラフで描く

df <- subset(get.input(), !is.na(time_s))
set.paper_style(mar = c(4.0, 4.2, 2.0, 0.8))
plot(df$time_s, df$displacement_m,
  type = "o", pch = 20, lwd = 1.6, col = "#1f77b4",
  xlab = "Time / s", ylab = "Displacement / m")
abline(h = 0, col = "#9ca3af", lty = 2)   # 中心線

カテゴリ比較の棒グラフ（誤差棒つき）

df <- subset(get.input(), label != "")
cols <- c("v_1", "v_2", "v_3")
df$mean <- rowMeans(df[cols], na.rm = TRUE)
df$sd   <- apply(df[cols], 1, sd, na.rm = TRUE)

set.paper_style(mar = c(4.6, 4.2, 2.0, 0.8))
bar_x <- barplot(df$mean, names.arg = df$label,
  col = "#cbd5e1", border = "#475569",
  ylim = c(0, max(df$mean + df$sd) * 1.15),
  xlab = "Category", ylab = "Value / unit")
arrows(bar_x, df$mean - df$sd, bar_x, df$mean + df$sd,
  angle = 90, code = 3, length = 0.05)
